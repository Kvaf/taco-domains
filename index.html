Developer Tools

RESTful API, webhooks, CLI tool, and SDKs for Node.js and Python. Automate everything.

Node.js Python cURL
// Install: npm install @taco-domains/sdk import { TacoClient } from '@taco-domains/sdk'; const taco = new TacoClient('YOUR_API_KEY'); // Search for available domains const results = await taco.search('my-awesome-taco', { tlds: ['.com', '.io', '.taco'], includePrice: true }); // Register a domain const domain = await taco.domains.register({ name: 'my-awesome-taco.com', years: 2, whoisPrivacy: true, autoRenew: true }); console.log(`🌮 Registered: ${domain.name}`);
# Install: pip install taco-domains from taco_domains import TacoClient taco = TacoClient(api_key="YOUR_API_KEY") # Search for available domains results = taco.search( query="my-awesome-taco", tlds=[".com", ".io", ".taco"] ) # Register a domain domain = taco.domains.register( name="my-awesome-taco.com", years=2, whois_privacy=True ) print(f"🌮 Registered: {domain.name}")
# Search domains curl -X GET "https://api.taco.domains/v1/search?q=my-awesome-taco" \ -H "Authorization: Bearer YOUR_API_KEY" # Register a domain curl -X POST "https://api.taco.domains/v1/domains/register" \ -H "Authorization: Bearer YOUR_API_KEY" \ -H "Content-Type: application/json" \ -d '{ "name": "my-awesome-taco.com", "years": 2, "whois_privacy": true }'
